with open("filename.txt", "r") as file:
    for line in file:
        # Split the line by '|' character
        parts = line.strip().split('|')
        if len(parts) == 2:
            # Assuming the second part is an integer value
            try:
                value = int(parts[1])
                if value > 10:
                    print("Value is greater than 10:", value)
            except ValueError:
                print("Invalid integer value on line:", line.strip())
        else:
            print("Invalid line format:", line.strip())
